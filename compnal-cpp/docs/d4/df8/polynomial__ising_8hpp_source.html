50 void SetInteraction(const std::int32_t degree, const RealType value) {
51 if (degree < 0) {
52 throw std::runtime_error("degree must be larger than or equal to 0");
53 }
54 if (std::abs(value) > std::numeric_limits<RealType>::epsilon()) {
55 interaction_.resize(degree + 1);
56 interaction_[degree] = value;
57 }
58 }

80 static RealType CalculateMoment(const std::vector<std::vector<OPType>> &samples, const std::int32_t degree) {
81 RealType val = RealType{0.0};
82 if (degree == 1) {
83#pragma omp parallel for schedule(guided) reduction(+: val)
84 for (std::int32_t i = 0; i < static_cast<std::int32_t>(samples.size()); ++i) {
85 const auto &sample = samples[i];
86 RealType temp = RealType{0.0};
87 for (std::int32_t j = 0; j < static_cast<std::int32_t>(sample.size()); ++j) {
88 temp += sample[j];
89 }
90 val += temp/sample.size();
91 }
92 }
93 else if (degree == 2) {
94#pragma omp parallel for schedule(guided) reduction(+: val)
95 for (std::int32_t i = 0; i < static_cast<std::int32_t>(samples.size()); ++i) {
96 const auto &sample = samples[i];
97 const std::int32_t sample_size = static_cast<std::int32_t>(sample.size());
98 RealType temp = RealType{0.0};
99 for (std::int32_t j_1 = 0; j_1 < sample_size; ++j_1) {
100 for (std::int32_t j_2 = 0; j_2 < sample_size; ++j_2) {
101 temp += sample[j_1]*sample[j_2];
102 }
103 }
104 val += temp/(sample_size*sample_size);
105 }
106 }
107 else if (degree == 4) {
108#pragma omp parallel for schedule(guided) reduction(+: val)
109 for (std::int32_t i = 0; i < static_cast<std::int32_t>(samples.size()); ++i) {
110 const auto &sample = samples[i];
111 const std::int32_t sample_size = static_cast<std::int32_t>(sample.size());
112 RealType temp = RealType{0.0};
113 for (std::int32_t j_1 = 0; j_1 < sample_size; ++j_1) {
114 for (std::int32_t j_2 = 0; j_2 < sample_size; ++j_2) {
115 for (std::int32_t j_3 = 0; j_3 < sample_size; ++j_3) {
116 for (std::int32_t j_4 = 0; j_4 < sample_size; ++j_4) {
117 temp += sample[j_1]*sample[j_2]*sample[j_3]*sample[j_4];
118 }
119 }
120 }
121 }
122 val += temp/(sample_size*sample_size*sample_size*sample_size);
123 }
124 }
125 else {
126 throw std::runtime_error("Under construction");
127 }
128 return val/samples.size();
129 }

131private:
132 std::vector<RealType> interaction_;

136 RealType energy = 0;
137 return energy;
138 }
