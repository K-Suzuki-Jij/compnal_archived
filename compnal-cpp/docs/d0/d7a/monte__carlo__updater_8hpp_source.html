33template<typename RealType, class ModelType>
34void ExecuteMetropolis(std::vector<utility::SpinType> *sample,
35 const ModelType &model,
36 const std::int32_t num_sweeps,
37 const RealType beta,
38 const std::uint64_t seed) {
39
40 //using RealType = typename ModelType::ValueType;
41
42 if (sample->size() != static_cast<std::size_t>(model.GetSystemSize())) {
43 throw std::runtime_error("The sample size is not equal to the system size.");
44 }
45
46 const std::int32_t system_size = model.GetSystemSize();
47
48 // Set random number engine
49 utility::RandType random_number_engine(seed);
50 std::uniform_real_distribution<RealType> dist_real(0, 1);
51 std::uniform_int_distribution<std::int32_t> dist_system_size(0, system_size - 1);
52
53 // Prepare system size index list (0, 1, 2, ..., N - 1)
54 std::vector<std::int32_t> system_size_index_list(system_size);
55 std::iota(system_size_index_list.begin(), system_size_index_list.end(), 0);
56
57 // Set energy difference
58 std::vector<RealType> energy_difference(system_size);
59 SetEnergyDifference<RealType>(&energy_difference, &system_size_index_list, *sample, model);
60
61 for (std::int32_t sweep_count = 0; sweep_count < num_sweeps; sweep_count++) {
62 for (std::int32_t i = 0; i < system_size; i++) {
63 const std::int32_t index = dist_system_size(random_number_engine);
64 if (energy_difference[index] <= 0.0 || std::exp(-beta*energy_difference[index]) > dist_real(random_number_engine)) {
65 UpdateConfiguration(sample, &energy_difference, &system_size_index_list, index, model);
66 }
67 }
68 }
69}
