111 void Run(const std::uint64_t seed) {
112 if (num_samples_ < 0) {
113 throw std::runtime_error("num_samples must be non-negative integer.");
114 }
115 if (num_sweeps_ < 0) {
116 throw std::runtime_error("num_sweeps must be non-negative integer.");
117 }
118 seed_ = seed;
119
120 samples_.resize(num_samples_);
121 for (std::size_t i = 0; i < samples_.size(); ++i) {
122 samples_[i].resize(model.GetSystemSize());
123 }
124 samples_.shrink_to_fit();
125
126 utility::RandType random_number_engine(seed_);
127 std::vector<std::uint64_t> configuration_seed_list(num_samples_);
128 std::vector<std::uint64_t> execute_seed_list(num_samples_);
129
130 for (int32_t i = 0; i < num_samples_; ++i) {
131 configuration_seed_list[i] = random_number_engine();
132 execute_seed_list[i] = random_number_engine();
133 }
134
135 if (this->cmc_updater == CMCUpdater::METROPOLIS) {
136#pragma omp parallel for schedule(guided)
137 for (std::int32_t sample_count = 0; sample_count < num_samples_; sample_count++) {
138 RandomizeConfiguration(&samples_[sample_count], configuration_seed_list[sample_count]);
139 updater::ExecuteMetropolis(&samples_[sample_count], model, num_sweeps_, beta_, execute_seed_list[sample_count]);
140 }
141 }
142 else if (this->cmc_updater == CMCUpdater::HEAT_BATH) {
143#pragma omp parallel for schedule(guided)
144 for (std::int32_t sample_count = 0; sample_count < num_samples_; sample_count++) {
145 RandomizeConfiguration(&samples_[sample_count], configuration_seed_list[sample_count]);
146 //updater::ExecuteHeatBath(&samples_[sample_count], model, num_sweeps_, beta_, execute_seed_list[sample_count]);
147 }
148 }
149 else {
150 throw std::runtime_error("Unknown Updater");
151 }
152
153 }

170 void RandomizeConfiguration(std::vector<utility::SpinType> *sample,
171 const std::uint64_t seed) const {
172 std::uniform_int_distribution<utility::SpinType> dist(0, 1);
173 utility::RandType random_number_engine(seed);
174 for (std::size_t i = 0; i < sample->size(); i++) {
175 (*sample)[i] = 2*dist(random_number_engine) - 1;
176 }
